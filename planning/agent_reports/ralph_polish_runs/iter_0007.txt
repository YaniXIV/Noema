func mixLast(out *fr.Element, in []fr.Element, m [][]fr.Element, idx int) {
	out.SetZero()
	for row := range in {
		var term fr.Element
		term.Mul(&m[row][idx], &in[row])
		out.Add(out, &term)
	}
}
